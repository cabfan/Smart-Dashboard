部门信息表：tem_dept，该表记录了松毅软件开发有限公司（松毅科技）的组织架构信息，是一个树型结构。它包含以下字段：
    - dept_id: 部门编号，主键；
    - parent_id: 父部门编号，通过它可以实现父子的树形结构；
    - dept_name：部门名称；
    - order_num：显示顺序；
    - leader：负责人；
    - phone：部门电话；
    - email：部门邮箱；
    - status：部门状态（0正常 1停用 2删除）；
    - create_by：创建者，指的是创建人的名称；
    - create_time：创建时间；
    - update_by：更新者，指的是更新人的名称；
    - update_time：更新时间；
    - remark：备注；
    - level：部门的级别，指的是层级，parent_id为0的是1，子级是父的level+1；


CREATE TABLE `tem_dept`  (
  `dept_id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '部门id',
  `parent_id` bigint(20) NULL DEFAULT 0 COMMENT '父部门id',
  `dept_name` varchar(30) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT '' COMMENT '部门名称',
  `order_num` int(4) NULL DEFAULT 0 COMMENT '显示顺序',
  `leader` varchar(20) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT NULL COMMENT '负责人',
  `phone` varchar(11) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT NULL COMMENT '联系电话',
  `email` varchar(50) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT NULL COMMENT '邮箱',
  `status` char(1) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT '0' COMMENT '部门状态（0正常 1停用 2删除）',
  `create_by` varchar(64) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT '' COMMENT '创建者',
  `create_time` datetime(0) NULL DEFAULT NULL COMMENT '创建时间',
  `update_by` varchar(64) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT '' COMMENT '更新者',
  `update_time` datetime(0) NULL DEFAULT NULL COMMENT '更新时间',
  `remark` varchar(500) CHARACTER SET utf8 COLLATE utf8_general_ci NULL DEFAULT '' COMMENT '备注',
  `level` int(11) NULL DEFAULT NULL COMMENT '部门级别',
  PRIMARY KEY (`dept_id`) USING BTREE
) ENGINE = InnoDB AUTO_INCREMENT = 1890227796720893954 CHARACTER SET = utf8 COLLATE = utf8_general_ci COMMENT = '部门表' ROW_FORMAT = Compact;


SELECT
    dept_id AS 部门编号,
    parent_id AS 父部门编号,
    dept_name AS 部门名称,
    order_num AS 顺序号,
    leader AS 负责人,
    phone AS 部门电话,
    email AS 部门邮箱,
    status AS 部门邮箱,
    create_by AS 创建人,
    create_time AS 创建时间,
    update_by AS 最后更信人,
    update_time AS 最后更新时间,
    remark AS 备注,
    level AS 级别
FROM
    tem_dept;
